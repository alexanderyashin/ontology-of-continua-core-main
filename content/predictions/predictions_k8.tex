\section{Predictions for \texorpdfstring{$K_8$}{K_8}}
\label{sec:predictions-k8}

Level $K_8$ describes \emph{civilizational-infrastructural continua}: 
large-scale systems whose stability depends on infrastructure, 
economic flows, technological bases, regulatory architectures, 
collective memory, and long-range coordination mechanisms.  
The defining structural objects of $K_8$ are:
\[
\Omega(K_8),\quad A^{8},\quad P^{8},\quad \Theta^{8},\quad 
J^{8},\quad C^{8},\quad k(K_8),\quad T_{\mathrm{infra}}.
\]

A system qualifies as a $K_8$ continuum when:
\[
C^{8} \neq 0,\qquad 
J^{8}_{\mathrm{infra}} \neq 0,\qquad
k(K_8) > 0,\qquad
\partial\Omega^{8} \ \text{defines macro-level institutional-infrastructural boundaries}.
\]

Below are the predictions grouped by structural class.

% ---------------------------------------------------------------
\subsection{P1: Predictions on the Birth of \texorpdfstring{$K_8$}{K_8}}

\paragraph{(P8.1) Infrastructure Attractor Formation.}
$K_8$ emerges when an attractor of infrastructure states appears:
\[
A_{\mathrm{infra}}^8 \neq 0 
\iff 
\text{persistent technical–economic–legal cycles form}.
\]

\paragraph{(P8.2) Threshold for Macro-structural Emergence.}
Let $P_{\mathrm{infra}}$ denote the infrastructure potential.  
A necessary condition:
\[
P_{\mathrm{infra}} > \Theta_{\mathrm{infra}},
\]
predicting a detectable phase transition from social continuity ($K_7$) 
to infrastructural invariants ($K_8$).

\paragraph{(P8.3) Persistence Condition.}
An emerging $K_8$ continuum persists only if long-range flows satisfy:
\[
J_{\mathrm{logistics}} > J_{\mathrm{min}},
\quad  
J_{\mathrm{regulation}} > 0,
\quad
J_{\mathrm{information}} \ \text{remains non-fragmented}.
\]

\paragraph{(P8.4) Civilizational Memory Formation.}
$K_8$ predicts that once collective memory becomes an axis:
\[
A_{\mathrm{memory}}^8 \neq 0,
\]
the continuum gains temporal depth (historical inertia).

% ---------------------------------------------------------------
\subsection{P2: Predictions about Axes and Structural Geometry}

\paragraph{(P8.5) Minimal Axes at \texorpdfstring{$K_8$}{K_8}.}
$K_8$ necessarily exhibits the following axes:
\[
A_{\mathrm{infra}},\ 
A_{\mathrm{tech}},\ 
A_{\mathrm{econ}},\ 
A_{\mathrm{reg}},\ 
A_{\mathrm{memory}}.
\]

\paragraph{(P8.6) Separation of Scales.}
$K_8$ predicts strong multi-scale separation:
\[
\tau_{\mathrm{tech}} \ll \tau_{\mathrm{institution}} \ll \tau_{\mathrm{civilizational}}.
\]

\paragraph{(P8.7) Existence of Regulatory Geometry.}
The geometry of constraints (laws, standards) behaves as:
\[
\partial\Omega_{\mathrm{reg}} 
\ \text{acts as a rigid boundary reducing} \ 
T_{\mathrm{infra}}.
\]

% ---------------------------------------------------------------
\subsection{P3: Predictions for Potentials $P^{8}$}

\paragraph{(P8.8) Economic Potential as Stabilizer.}
$P_{\mathrm{econ}}$ satisfies:
\[
\frac{d}{dt} k(K_8) 
= F(J_{\mathrm{trade}}, C_{\mathrm{production}}, \Theta_{\mathrm{econ}}).
\]

\paragraph{(P8.9) Technological Acceleration.}
If:
\[
P_{\mathrm{tech}} \uparrow,
\]
then:
\[
\Theta_{\mathrm{infra}} \downarrow,
\]
predicting easier formation of new infrastructural attractors.

\paragraph{(P8.10) Collective Memory Inertia.}
Potential $P_{\mathrm{memory}}$ increases stability but slows adaptation:
\[
P_{\mathrm{memory}} \uparrow 
\Rightarrow 
\frac{d}{dt} \Theta_{\mathrm{institution}} \uparrow.
\]

\paragraph{(P8.11) Regulatory Potential $P_{\mathrm{reg}}$.}
$K_8$ predicts that:
\[
P_{\mathrm{reg}} > \Theta_{\mathrm{reg}}
\]
is required for large-scale coordination.

% ---------------------------------------------------------------
\subsection{P4: Predictions for Infrastructural and Economic Flows}

\paragraph{(P8.12) Asymmetric Infrastructure Flows.}
$K_8$ predicts intrinsic directionality:
\[
J_{\mathrm{infra}}(x \to y) 
\neq 
J_{\mathrm{infra}}(y \to x),
\]
leading to core–periphery structures.

\paragraph{(P8.13) Logistic Thresholds.}
Critical flows:
\[
J_{\mathrm{logistics}} > \Theta_{\mathrm{logistics}}
\]
are necessary for the survival of the continuum.

\paragraph{(P8.14) Regulatory Flow Closure.}
A functioning $K_8$ always satisfies:
\[
J_{\mathrm{regulation}}: \text{policy} 
\to \text{implementation} 
\to \text{feedback loop}.
\]

\paragraph{(P8.15) Energy and Resource Scaling Laws.}
Civilizational energy flow:
\[
J_{\mathrm{energy}} \sim \Omega_{\mathrm{population}}^\alpha,
\qquad 
\alpha > 1,
\]
predicting superlinear scaling.

% ---------------------------------------------------------------
\subsection{P5: Predictions for Cycles $C^{8}$}

\paragraph{(P8.16) Production–Distribution–Consumption Cycle.}
$K_8$ predicts a self-sustaining cycle:
\[
C_{\mathrm{econ}}
: \ 
\text{production} 
\to \text{distribution} 
\to \text{consumption} 
\to \text{renewal}.
\]

\paragraph{(P8.17) Institutional Reproduction Cycle.}
Institutions must follow:
\[
C_{\mathrm{institution}}^8 
=
\text{codification} 
\to \text{enforcement} 
\to \text{adaptation}.
\]

\paragraph{(P8.18) Innovation Cycle.}
$K_8$ predicts:
\[
C_{\mathrm{innovation}} 
: P_{\mathrm{tech}} 
\to \text{application} 
\to J_{\mathrm{econ}} 
\to P_{\mathrm{tech}}',
\]
forming positive feedback.

\paragraph{(P8.19) Memory Cycle.}
Collective memory evolves through:
\[
\text{selection} 
\to \text{institutional embedding} 
\to \text{reproduction}.
\]

% ---------------------------------------------------------------
\subsection{P6: Predictions for Stability and k(K₈)}

\paragraph{(P8.20) Multi-Threshold Stability Condition.}
\[
k(K_8) > 0 
\iff 
\forall k: \ P^8 \in \mathcal{D}_k^{(8)},
\]
i.e.\ all infrastructural, economic, regulatory and memory thresholds must hold.

\paragraph{(P8.21) Stress Response.}
If infrastructural tension:
\[
T_{\mathrm{infra}} > \Theta_{\mathrm{stress}}^{(8)},
\]
the system predicts either:
\begin{itemize}
    \item infrastructure branching,
    \item regulatory mutation,
    \item collapse of logistic flows.
\end{itemize}

\paragraph{(P8.22) Predictable Failure Modes.}
K8 collapses when:
\[
J_{\mathrm{energy}} = 0,
\quad \text{or} \quad
J_{\mathrm{logistics}} = 0,
\quad \text{or} \quad
J_{\mathrm{regulation}} = 0.
\]

\paragraph{(P8.23) Predictable Polarisation at Civilizational Scale.}
If:
\[
P_{\mathrm{memory}} \uparrow \ \text{and} \ 
J_{\mathrm{information}} \downarrow,
\]
then:
\[
\text{polarisation attractors emerge}.
\]

% ---------------------------------------------------------------
\subsection{P7: Predictions Concerning Collapse of \texorpdfstring{$K_8$}{K_8}}

\paragraph{(P8.24) Collapse via Infrastructure Decay.}
If:
\[
C_{\mathrm{infra}}^8 = 0,
\]
the continuum loses its boundary and collapses.

\paragraph{(P8.25) Collapse via Economic Cycle Failure.}
If:
\[
C_{\mathrm{econ}} = 0,
\]
the system enters $k(K_8) \to 0$.

\paragraph{(P8.26) Collapse via Memory Fragmentation.}
If:
\[
A_{\mathrm{memory}}^8 \ \text{splits into incompatible subspaces},
\]
the continuum fragments into sub-$K_8$ entities.

\paragraph{(P8.27) Collapse via Regulatory Failure.}
If:
\[
P_{\mathrm{reg}} < \Theta_{\mathrm{reg}},
\]
long-range coordination fails and $\Omega(K_8)$ dissolves.

% ---------------------------------------------------------------
\subsection{P8: Predictions for Transition to \texorpdfstring{$K_9$}{K_9}}

\paragraph{(P8.28) Meta-structural Layer Emergence.}
Transition to $K_9$ begins when:
\[
A_{\mathrm{meta}}^9 \neq 0,
\]
i.e.\ when the system constructs theories, models, and meta-representations of itself.

\paragraph{(P8.29) Threshold of Abstract Coherence.}
$K_8 \to K_9$ transition requires:
\[
P_{\mathrm{abstraction}} > \Theta_{\mathrm{meta}},
\]
predicting emergence of philosophical, scientific, mathematical structures.

\paragraph{(P8.30) Recursive Institutionalisation.}
When:
\[
C_{\mathrm{institution}}^8 
\text{ includes self-description},
\]
a new dimension opens and $K_9$ is born.

% ---------------------------------------------------------------
\subsection{Summary}

The predictions for $K_8$ include:
\begin{itemize}
    \item formation of infrastructural attractors,
    \item multi-axis macro-organization (technical, economic, regulatory, memory),
    \item non-trivial potentials governing civilizational stability,
    \item superlinear energy scaling laws,
    \item closure of economic, institutional, innovation and memory cycles,
    \item predictable stress responses, polarisation, and collapse modes,
    \item emergence of meta-structures signalling the transition to $K_9$.
\end{itemize}
